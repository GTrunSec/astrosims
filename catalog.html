Python Query

Example python code to apply the above filters and retrieve data.
To use, download and install this module. 

import fi_astrosims.client
gaea = fi_astrosims.client.Simulation("gaea", host = "http://localhost:8092")
q = gaea.query(fields = ["PPos_x", "PPos_y", "z", "z_pos", "Pos_x", "Pos_y", "Pos_z", "Vel_x", "Vel_y", "Vel_z", "Type", "GalID", "MMSubID", "CentralMvir", "Mvir", "Rvir", "Vvir", "Vmax", "StellarMass", "ColdGas", "HI", "GasDiskRadius", "StellarDiskRadius", "GasSpin_x", "GasSpin_y", "GasSpin_z"], z_pos = (0.0016757736942307714, 0.03066790990567455))
dat = q.numpy()
